# 1.29

  • sms list에서 메세지 필터( sms, lms, 전체 ) 기능 추가 (디자인 안나옴)
    • watch 개념 사용
<template>
  <el-radio-group v-model="smsType">
    <el-radio
      v-for="option in smsTypeOptions"
      :key="option.label"
      :label="option.value"
    >
      {{ option.label }}
    </el-radio>
  </el-radio-group>
</template>

// 위와 같이 라디오는 v-model은 smsType을 따른 다고 할 때,

<script>
export default {
  data() {
    return {
      smsType: "ALL",
      smsTypeOptions: [
        { value: "ALL", label: "전체" },
        { value: "SMS", label: "SMS" },
        { value: "LMS", label: "LMS" }
      ]
    };
  },
  watch: {
    smsType() {
      this.getMessages();
    }
  }
  //watch로 smsType변경을 감지하고 있으니 라디오가 변경 될 때마다
  //this.getMessages 함수가 실행된다.
};
</script>
